def format_title(title):
  sep_index = title.find(':') if title.find(':') != -1 else title.find('-')
  if sep_index != -1:
    title = title[:sep_index]
  return title[:10]

app_reviews = []

for ap in tqdm(app_packages):
    for score in list(range(1, 6)):
        for sort_order in [Sort.RATING, Sort.NEWEST]:
            rvs, _ = reviews(
            ap,
            lang='en',
            #country='us',
            sort=sort_order,
            count= 900 if score == 3 else 100,
            filter_score_with=score
          )
        for r in rvs:
            r['sortOrder'] = 'rating' if sort_order == Sort.RATING else 'newest'
            r['appId'] = ap
            app_reviews.extend(rvs)
